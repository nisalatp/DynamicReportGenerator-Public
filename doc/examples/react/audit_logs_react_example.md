# Audit Logs Example: React (Hooks)

This example demonstrates how to view the execution and lifecycle audit logs generated by the Dynamic Report Generator engine. Every significant event — report creation, execution, assignment, deletion, and errors — is automatically recorded.

## The Frontend Implementation

```jsx
import React, { useState, useEffect } from 'react';

export default function AuditLogViewer() {
  const [logs, setLogs] = useState([]);
  const [filters, setFilters] = useState({
    action: '',        // 'created', 'executed', 'assigned', 'updated', 'deleted', 'error'
    reportId: '',
    userId: ''
  });
  const [loading, setLoading] = useState(false);

  const fetchLogs = async () => {
    setLoading(true);
    try {
      const params = new URLSearchParams();
      if (filters.action) params.append('action', filters.action);
      if (filters.reportId) params.append('report_id', filters.reportId);
      if (filters.userId) params.append('user_id', filters.userId);

      const response = await fetch(`/api/admin/logs?${params.toString()}`);
      setLogs(await response.json());
    } catch (error) {
      console.error("Failed to fetch logs", error);
    } finally {
      setLoading(false);
    }
  };

  useEffect(() => { fetchLogs(); }, []);

  const getActionBadgeClass = (action) => {
    switch (action) {
      case 'executed': return 'badge-success';
      case 'error': return 'badge-danger';
      case 'deleted': return 'badge-warning';
      case 'created': return 'badge-info';
      case 'assigned': return 'badge-purple';
      case 'unassigned': return 'badge-orange';
      default: return 'badge-default';
    }
  };

  return (
    <div className="audit-logs">
      <h2>Audit Log Viewer</h2>

      {/* Filters */}
      <div className="filter-bar">
        <select 
          value={filters.action} 
          onChange={e => setFilters({ ...filters, action: e.target.value })}
        >
          <option value="">All Actions</option>
          <option value="created">Created</option>
          <option value="updated">Updated</option>
          <option value="executed">Executed</option>
          <option value="assigned">Assigned</option>
          <option value="unassigned">Unassigned</option>
          <option value="deleted">Deleted</option>
          <option value="error">Errors</option>
        </select>

        <input 
          type="text" 
          placeholder="Report ID"
          value={filters.reportId}
          onChange={e => setFilters({ ...filters, reportId: e.target.value })}
        />

        <input 
          type="text" 
          placeholder="User ID"
          value={filters.userId}
          onChange={e => setFilters({ ...filters, userId: e.target.value })}
        />

        <button onClick={fetchLogs} disabled={loading}>
          {loading ? 'Loading...' : 'Search'}
        </button>
      </div>

      {/* Log Table */}
      <table className="log-table">
        <thead>
          <tr>
            <th>Timestamp</th>
            <th>Action</th>
            <th>Report</th>
            <th>User</th>
            <th>Details</th>
          </tr>
        </thead>
        <tbody>
          {logs.map(log => (
            <tr key={log.id} className={log.action === 'error' ? 'row-error' : ''}>
              <td>{new Date(log.created_at).toLocaleString()}</td>
              <td>
                <span className={`badge ${getActionBadgeClass(log.action)}`}>
                  {log.action}
                </span>
              </td>
              <td>{log.report_name || log.saved_report_id || '—'}</td>
              <td>{log.user_name || log.user_id || 'System'}</td>
              <td>
                {log.action === 'error' && (
                  <details>
                    <summary>View Error</summary>
                    <pre className="error-trace">{log.details}</pre>
                  </details>
                )}
                {log.action === 'executed' && (
                  <span>Rows: {log.row_count || 'N/A'}</span>
                )}
                {log.action === 'assigned' && (
                  <span>To User #{log.details}</span>
                )}
              </td>
            </tr>
          ))}
        </tbody>
      </table>

      {logs.length === 0 && !loading && (
        <p className="empty-state">No audit logs found matching your criteria.</p>
      )}
    </div>
  );
}
```

## Backend Controller Example

```php
use Nisalatp\DynamicReportGenerator\Models\ReportLog;

public function index(Request $request)
{
    $query = ReportLog::with(['savedReport', 'user'])
        ->orderBy('created_at', 'desc');

    if ($request->filled('action')) {
        $query->where('action', $request->input('action'));
    }
    if ($request->filled('report_id')) {
        $query->where('saved_report_id', $request->input('report_id'));
    }
    if ($request->filled('user_id')) {
        $query->where('user_id', $request->input('user_id'));
    }

    return response()->json(
        $query->limit(100)->get()->map(fn ($log) => [
            'id' => $log->id,
            'action' => $log->action,
            'saved_report_id' => $log->saved_report_id,
            'report_name' => $log->savedReport?->name,
            'user_id' => $log->user_id,
            'user_name' => $log->user?->name,
            'details' => $log->details,
            'row_count' => $log->row_count,
            'created_at' => $log->created_at,
        ])
    );
}
```

> [!NOTE]
> The `dynamic_report_logs` table automatically records: `created`, `updated`, `executed`, `assigned`, `unassigned`, `deleted`, and `error` actions. Error logs include the exception message and stack trace for debugging.
