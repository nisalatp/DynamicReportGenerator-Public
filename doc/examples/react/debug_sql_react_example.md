# Debug SQL & Join Plan Example: React (Hooks)

This example demonstrates how to inspect the raw compiled SQL and the BFS join plan generated by the engine. This is essential for developers debugging complex multi-table reports and for AI agents that need to verify their generated AST payloads.

## The Frontend Implementation

```jsx
import React, { useState } from 'react';

export default function SqlDebugger() {
  const [rawSql, setRawSql] = useState('');
  const [joinPlan, setJoinPlan] = useState(null);
  const [loading, setLoading] = useState(false);

  const payload = {
    baseModel: 'User',
    targetModels: ['Order', 'Product'],
    selectedAttributes: [
      { modelClass: 'User', column: 'name', type: 'string' },
      { modelClass: 'Product', column: 'name', type: 'string', alias: 'product_name' }
    ],
    groupBys: [
      { attribute: { modelClass: 'User', column: 'country', type: 'string' } }
    ],
    aggregates: [
      {
        attribute: { modelClass: 'Order', column: 'amount', type: 'integer' },
        function: 'SUM',
        alias: 'total_revenue'
      }
    ],
    innerFilters: {
      type: 'group',
      logic: 'and',
      children: [
        {
          type: 'leaf',
          attribute: { modelClass: 'User', column: 'status', type: 'string' },
          operator: '=',
          value: 'active'
        }
      ]
    },
    outerFilters: null,
    sorts: [
      { attribute: { modelClass: 'Order', column: 'total_revenue', isVirtual: true }, direction: 'DESC' }
    ]
  };

  // ─── Get Raw SQL ────────────────────────────────────────────
  const fetchRawSql = async () => {
    setLoading(true);
    try {
      const response = await fetch('/api/report/debug/sql', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify(payload)
      });
      const data = await response.json();
      setRawSql(data.sql);
    } catch (error) {
      console.error("Failed to fetch raw SQL", error);
    } finally {
      setLoading(false);
    }
  };

  // ─── Get BFS Join Plan ──────────────────────────────────────
  const fetchJoinPlan = async () => {
    setLoading(true);
    try {
      const response = await fetch('/api/report/debug/join-plan', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify(payload)
      });
      const data = await response.json();
      setJoinPlan(data);
    } catch (error) {
      console.error("Failed to fetch join plan", error);
    } finally {
      setLoading(false);
    }
  };

  return (
    <div className="sql-debugger">
      <h2>SQL Debugger & Join Plan Inspector</h2>

      <div className="actions">
        <button onClick={fetchRawSql} disabled={loading}>
          {loading ? 'Loading...' : 'Show Raw SQL'}
        </button>
        <button onClick={fetchJoinPlan} disabled={loading}>
          {loading ? 'Loading...' : 'Show BFS Join Plan'}
        </button>
      </div>

      {rawSql && (
        <div className="debug-panel">
          <h3>Compiled SQL</h3>
          <pre className="sql-output">{rawSql}</pre>
        </div>
      )}

      {joinPlan && (
        <div className="debug-panel">
          <h3>BFS Join Plan</h3>
          <table>
            <thead>
              <tr>
                <th>From</th>
                <th>To</th>
                <th>Type</th>
                <th>Direction</th>
                <th>Local Key</th>
                <th>Foreign Key</th>
              </tr>
            </thead>
            <tbody>
              {joinPlan.steps.map((step, i) => (
                <tr key={i}>
                  <td>{step.fromModel}</td>
                  <td>{step.toModel}</td>
                  <td>{step.relationType}</td>
                  <td>
                    <span className={`badge ${step.direction}`}>
                      {step.direction}
                    </span>
                  </td>
                  <td><code>{step.localKey}</code></td>
                  <td><code>{step.foreignKey}</code></td>
                </tr>
              ))}
            </tbody>
          </table>
        </div>
      )}
    </div>
  );
}
```

## Backend Controller Example

```php
use Nisalatp\DynamicReportGenerator\Facades\DynamicReport;
use Nisalatp\DynamicReportGenerator\Http\Requests\ReportBuilderRequest;

// Raw SQL
public function debugSql(Request $request)
{
    $reportRequest = ReportBuilderRequest::fromPayload($request->all());
    $query = DynamicReport::generate($reportRequest);
    
    return response()->json([
        'sql' => DynamicReport::toRawSql($query)
    ]);
}

// BFS Join Plan
public function debugJoinPlan(Request $request)
{
    $reportRequest = ReportBuilderRequest::fromPayload($request->all());
    $joinPlan = DynamicReport::explainJoinPlan($reportRequest);
    
    return response()->json($joinPlan);
}
```

## Example Output

### Raw SQL
```sql
SELECT t0.name, t0.country, t2.name AS product_name, SUM(t1.amount) AS total_revenue
FROM users AS t0
LEFT JOIN orders AS t1 ON t0.id = t1.user_id
LEFT JOIN products AS t2 ON t1.product_id = t2.id
WHERE t0.status = 'active'
GROUP BY t0.country
ORDER BY total_revenue DESC
LIMIT 5000
```

### BFS Join Plan
| From | To | Type | Direction | Local Key | Foreign Key |
|------|-----|------|-----------|-----------|-------------|
| User | Order | HasMany | forward | id | user_id |
| Order | Product | BelongsToMany | forward | id | order_id |

> [!TIP]
> The `direction` field in each `JoinStep` tells you whether the relationship was explicitly declared on the model (`forward`) or automatically synthesized by the engine's reverse inference (`reverse`).
