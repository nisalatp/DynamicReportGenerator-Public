# Audit Logs Example: Vue 3 (Composition API)

This example demonstrates how to view the execution and lifecycle audit logs generated by the Dynamic Report Generator.

## The Frontend Implementation

```vue
<template>
  <div class="audit-logs">
    <h2>Audit Log Viewer</h2>

    <div class="filter-bar">
      <select v-model="filters.action">
        <option value="">All Actions</option>
        <option value="created">Created</option>
        <option value="updated">Updated</option>
        <option value="executed">Executed</option>
        <option value="assigned">Assigned</option>
        <option value="unassigned">Unassigned</option>
        <option value="deleted">Deleted</option>
        <option value="error">Errors</option>
      </select>

      <input v-model="filters.reportId" type="text" placeholder="Report ID" />
      <input v-model="filters.userId" type="text" placeholder="User ID" />

      <button @click="fetchLogs" :disabled="loading">
        {{ loading ? 'Loading...' : 'Search' }}
      </button>
    </div>

    <table class="log-table">
      <thead>
        <tr>
          <th>Timestamp</th><th>Action</th><th>Report</th><th>User</th><th>Details</th>
        </tr>
      </thead>
      <tbody>
        <tr v-for="log in logs" :key="log.id" :class="{ 'row-error': log.action === 'error' }">
          <td>{{ new Date(log.created_at).toLocaleString() }}</td>
          <td><span :class="['badge', badgeClass(log.action)]">{{ log.action }}</span></td>
          <td>{{ log.report_name || log.saved_report_id || '—' }}</td>
          <td>{{ log.user_name || log.user_id || 'System' }}</td>
          <td>
            <details v-if="log.action === 'error'">
              <summary>View Error</summary>
              <pre class="error-trace">{{ log.details }}</pre>
            </details>
            <span v-else-if="log.action === 'executed'">Rows: {{ log.row_count || 'N/A' }}</span>
            <span v-else-if="log.action === 'assigned'">To User #{{ log.details }}</span>
          </td>
        </tr>
      </tbody>
    </table>

    <p v-if="logs.length === 0 && !loading" class="empty-state">No logs found.</p>
  </div>
</template>

<script setup>
import { ref, onMounted } from 'vue';
import axios from 'axios';

const logs = ref([]);
const loading = ref(false);
const filters = ref({ action: '', reportId: '', userId: '' });

const fetchLogs = async () => {
  loading.value = true;
  try {
    const params = {};
    if (filters.value.action) params.action = filters.value.action;
    if (filters.value.reportId) params.report_id = filters.value.reportId;
    if (filters.value.userId) params.user_id = filters.value.userId;
    const res = await axios.get('/api/admin/logs', { params });
    logs.value = res.data;
  } finally { loading.value = false; }
};

const badgeClass = (action) => ({
  executed: 'badge-success', error: 'badge-danger', deleted: 'badge-warning',
  created: 'badge-info', assigned: 'badge-purple', unassigned: 'badge-orange'
}[action] || 'badge-default');

onMounted(fetchLogs);
</script>
```

> [!NOTE]
> The `dynamic_report_logs` table automatically records: `created`, `updated`, `executed`, `assigned`, `unassigned`, `deleted`, and `error` actions.
